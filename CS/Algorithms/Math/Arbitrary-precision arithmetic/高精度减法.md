# 高精度减法

## [AcWing 792. 高精度减法](https://www.acwing.com/problem/content/description/794/)

给定两个正整数（不含前导 0)，计算它们的差，计算结果可能为负数。

### **输入格式**

共两行，每行包含一个整数。

### **输出格式**

共一行，包含所求的差。

### **数据范围**

$1$ ≤ 整数长度 ≤ $10^5$

### **输入样例：**

```
32
11
```

### **输出样例：**

```
21
```

---

### 不压位代码

```
#include <bits/stdc++.h>

using namespace std;

bool cmp(vector<int> &A, vector<int> &B) {
    if (A.size() != B.size()) return A.size() > B.size();
    for (int i = (int) A.size() - 1; i >= 0; i--) {
        if (A[i] != B[i]) return A[i] > B[i];
    }
    return true;
}

vector<int> sub(vector<int> &A, vector<int> &B) {
    vector<int> C;
    for (int i = 0, t = 0; i < A.size(); i++) {
        t = A[i] - t;
        if (i < B.size()) t -= B[i];
        C.push_back((t + 10) % 10);
        if (t < 0) t = 1;
        else t = 0;
    }

    while (C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}

int main() {
    string a, b;
    vector<int> A, B;
    cin >> a >> b;
    for (int i = (int) a.size() - 1; i >= 0; i--) A.push_back(a[i] - '0');
    for (int i = (int) b.size() - 1; i >= 0; i--) B.push_back(b[i] - '0');

    vector<int> C;

    if (cmp(A, B)) C = sub(A, B);
    else C = sub(B, A), cout << '-';

    for (int i = (int) C.size() - 1; i >= 0; i--) cout << C[i];
    cout << endl;

    return 0;
}
```

### 压9位的代码

```cpp
#include <bits/stdc++.h>

using namespace std;

const int mod = 1e9;

vector<int> get_num(string &a) {
    vector<int> A;
    for (int i = (int) a.size() - 1, s = 0, t = 1, j = 0; i >= 0; i--) {
        s += (a[i] - '0') * t, j++, t *= 10;
        if (j == 9 || i == 0) {
            A.push_back(s);
            j = 0, t = 1, s = 0;
        }
    }
    return A;
}

bool cmp(vector<int> A, vector<int> B) {
    if (A.size() != B.size()) return A.size() > B.size();
    for (int i = (int) A.size() - 1; i >= 0; i--) {
        if (A[i] != B[i]) return A[i] > B[i];
    }
    return true;
}

vector<int> sub(vector<int> A, vector<int> B) {
    vector<int> C;
    int t = 0;
    for (int i = 0; i <= A.size() - 1; i++) {
        t += A[i];
        if (i < B.size()) t -= B[i];
        C.push_back((t + mod) % mod);
        if (t < 0) t = -1;
        else t = 0;
    }
    while (C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}

int main() {
    string a, b;
    cin >> a >> b;

    auto A = get_num(a), B = get_num(b);

    if (cmp(A, B)) {
        auto C = sub(A, B);
        cout << C.back();
        for (int i = (int) C.size() - 2; i >= 0; i--) printf("%09d", C[i]);
    } else {
        printf("-");
        auto C = sub(B, A);
        cout << C.back();
        for (int i = (int) C.size() - 2; i >= 0; i--) printf("%09d", C[i]);
    }
    return 0;
}
```
